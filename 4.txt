# Function to find the waiting time for all processes
def findWaitingTime(processes, n, wt):
    rt = [0] * n  # Remaining burst times
    for i in range(n):
        rt[i] = processes[i][1]  # Burst time

    complete = 0
    t = 0
    minm = float('inf')
    short = 0
    check = False

    while complete != n:
        for j in range(n):
            if (processes[j][2] <= t and rt[j] < minm and rt[j] > 0):
                minm = rt[j]
                short = j
                check = True

        if not check:
            t += 1
            continue

        # Reduce remaining time
        rt[short] -= 1
        minm = rt[short]
        if minm == 0:
            minm = float('inf')

        # If a process gets completely executed
        if rt[short] == 0:
            complete += 1
            check = False
            fint = t + 1
            wt[short] = fint - processes[short][1] - processes[short][2]
            if wt[short] < 0:
                wt[short] = 0

        t += 1


# Function to calculate turnaround time
def findTurnAroundTime(processes, n, wt, tat):
    for i in range(n):
        tat[i] = processes[i][1] + wt[i]


# Function to calculate average times and display results
def findavgTime(processes, n):
    wt = [0] * n
    tat = [0] * n

    findWaitingTime(processes, n, wt)
    findTurnAroundTime(processes, n, wt, tat)

    total_wt = 0
    total_tat = 0

    print("Processes\tBurst Time\tWaiting Time\tTurn-Around Time")
    for i in range(n):
        total_wt += wt[i]
        total_tat += tat[i]
        print(f"{processes[i][0]}\t\t{processes[i][1]}\t\t{wt[i]}\t\t{tat[i]}")

    print(f"\nAverage waiting time = {total_wt / n:.2f}")
    print(f"Average turn around time = {total_tat / n:.2f}")
